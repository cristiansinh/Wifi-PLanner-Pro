# WiFi Planner Pro v3.0 — PROFESSIONAL EDITION
# Dependencias de Python
# ============================================

# REQUERIDAS (instalación mínima):
pywebview>=4.0.0,<5.0.0
# Framework para ventanas nativas con HTML/CSS/JS
# Windows: Usa Edge WebView2 (incluido en Win10/11)
# macOS: Usa Safari WebView
# Linux: Usa GTK/WebKit

# OPCIONALES PERO MUY RECOMENDADAS:

pymupdf>=1.23.0
# Soporte PDF - permite cargar PDFs directamente
# Convierte páginas PDF a imágenes de alta calidad
# SIN ESTO: solo funcionarán imágenes (PNG, JPG, BMP)

pillow>=10.0.0
# Optimización de imágenes
# Redimensionamiento automático
# Conversión de formatos
# SIN ESTO: imágenes grandes pueden causar lag

psutil>=5.9.0
# Detección de recursos del sistema (CPU, RAM)
# Ajuste automático de calidad de rendering
# SIN ESTO: usará valores por defecto conservadores

# COMPILACIÓN (solo si vas a crear .exe):
pyinstaller>=6.0.0
# O alternativamente:
# auto-py-to-exe>=2.40.0
